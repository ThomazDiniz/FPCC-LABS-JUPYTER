theta1 <- function(d, s) {
    m = mean(d[s == "m"])
    f = mean(d[s == "f"])
    m - f
}

diferencas = iats %>% 
    group_by(referrer) %>% 
    summarise(diff = theta1(d_art, sex)) 
diferencas %>% 
    ggplot(aes(x = reorder(referrer, diff), y = diff)) +

Comparação via ICs

library(boot)

theta <- function(d, i) {
    agrupado = d %>% 
        slice(i) %>% 
        group_by(sex) %>% 
        summarise(media = mean(d_art))
    m = agrupado %>% filter(sex == "m") %>% pull(media)
    f = agrupado %>% filter(sex == "f") %>% pull(media)
    m - f
}

nested = iats %>% 
    group_by(referrer) %>% 
    mutate(n = n()) %>% 
    filter(n > 10) %>% 
    ungroup() %>% 
    nest(-referrer)
